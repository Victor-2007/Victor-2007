1) Distance-based “interest” flagging in Excel (Apache POI) 
package excelOperations;

import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;

import org.apache.poi.ss.usermodel.CellType;
import org.apache.poi.xssf.usermodel.*;

/**
 * Reads a single Excel sheet containing two sets of 3D coordinates (e.g., Molecule B and Molecule C).
 * For each Molecule B row, it checks whether any Molecule C row lies within a threshold distance.
 * Writes an "Interest" column (Yes/No) for Molecule B rows.
 */
public class ReadingExcel {

  public static void main(String[] args) throws IOException {
    // Usage: java ReadingExcel input.xlsx
    if (args.length < 1) {
      System.out.println("Usage: java ReadingExcel ");
      return;
    }

    String excelFilePath = args[0];
    double threshold = 10.0; // distance cutoff

    try (FileInputStream in = new FileInputStream(excelFilePath);
         XSSFWorkbook workbook = new XSSFWorkbook(in)) {

      XSSFSheet sheet = workbook.getSheetAt(0);

      // Add Interest header in column 13 (0-indexed = 12)
      XSSFRow headerRow = sheet.getRow(0);
      XSSFCell headerCell = headerRow.createCell(12, CellType.STRING);
      headerCell.setCellValue("Interest");

      int lastRow = sheet.getLastRowNum();

      // Example layout:
      // Rows 1..869  = Molecule B
      // Rows 870..lastRow = Molecule C
      for (int i = 1; i < 870; i++) {
        XSSFRow rowB = sheet.getRow(i);
        double xB = rowB.getCell(6).getNumericCellValue();
        double yB = rowB.getCell(7).getNumericCellValue();
        double zB = rowB.getCell(8).getNumericCellValue();

        boolean isOfInterest = false;
        for (int j = 870; j <= lastRow; j++) {
          XSSFRow rowC = sheet.getRow(j);
          double xC = rowC.getCell(6).getNumericCellValue();
          double yC = rowC.getCell(7).getNumericCellValue();
          double zC = rowC.getCell(8).getNumericCellValue();

          double dx = xC - xB;
          double dy = yC - yB;
          double dz = zC - zB;
          double distance = Math.sqrt(dx*dx + dy*dy + dz*dz);

          if (distance < threshold) {
            isOfInterest = true;
            break;
          }
        }

        XSSFCell interestCell = rowB.createCell(12, CellType.STRING);
        interestCell.setCellValue(isOfInterest ? "Yes" : "No");
      }

      // Overwrite the file in-place
      try (FileOutputStream out = new FileOutputStream(excelFilePath)) {
        workbook.write(out);
      }
    }

    System.out.println("Done");
  }
}

This script reads 3D coordinate columns for two molecule groups on a single sheet. For each row in the first group, it scans the second group and flags the row as “Yes” if any point is within a distance threshold (10 Å here). The output is written back into an “Interest” column so downstream steps can filter and count contacts without manual review.

package excelOperations;

import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;

import org.apache.poi.ss.usermodel.CellType;
import org.apache.poi.xssf.usermodel.*;

/**
 * Similar to the first script, but writes the Interest flag for Molecule C rows
 * by checking whether any Molecule B row is within the threshold.
 * Writes results to a new file (output path).
 */
public class SecondMoleculeInterest {

  public static void main(String[] args) throws IOException {
    // Usage: java SecondMoleculeInterest input.xlsx output.xlsx
    if (args.length < 2) {
      System.out.println("Usage: java SecondMoleculeInterest  ");
      return;
    }

    String inputPath = args[0];
    String outputPath = args[1];
    double threshold = 10.0;

    try (FileInputStream in = new FileInputStream(inputPath);
         XSSFWorkbook workbook = new XSSFWorkbook(in)) {

      XSSFSheet sheet = workbook.getSheetAt(0);
      int lastRow = sheet.getLastRowNum();

      // Rows 1..869 = Molecule B; rows 870..lastRow = Molecule C
      for (int i = 870; i <= lastRow; i++) {
        XSSFRow rowC = sheet.getRow(i);
        double xC = rowC.getCell(6).getNumericCellValue();
        double yC = rowC.getCell(7).getNumericCellValue();
        double zC = rowC.getCell(8).getNumericCellValue();

        boolean isOfInterest = false;
        for (int j = 1; j < 870; j++) {
          XSSFRow rowB = sheet.getRow(j);
          double xB = rowB.getCell(6).getNumericCellValue();
          double yB = rowB.getCell(7).getNumericCellValue();
          double zB = rowB.getCell(8).getNumericCellValue();

          double dx = xB - xC;
          double dy = yB - yC;
          double dz = zB - zC;
          double distance = Math.sqrt(dx*dx + dy*dy + dz*dz);

          if (distance < threshold) {
            isOfInterest = true;
            break;
          }
        }

        XSSFCell interestCell = rowC.getCell(12);
        if (interestCell == null) interestCell = rowC.createCell(12, CellType.STRING);
        interestCell.setCellValue(isOfInterest ? "Yes" : "No");
      }

      try (FileOutputStream out = new FileOutputStream(outputPath)) {
        workbook.write(out);
      }
    }

    System.out.println("Wrote updated file.");
  }
}

This variant computes the same proximity rule, but stores the “Interest” label on the second molecule group. That makes it easy to compute contact statistics for either side (e.g., percent of Molecule C residues near Molecule B).

3) Aggregating across multiple models (probability-style summary) 
package excelOperations;

import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.util.HashMap;
import java.util.Map;

import org.apache.poi.ss.usermodel.CellType;
import org.apache.poi.xssf.usermodel.*;

/**
 * Loads multiple model output files, counts how many models mark each row as Interest="Yes",
 * and writes a compact summary into a master sheet.
 */
public class ProbabilityAggregator {

  public static void main(String[] args) throws IOException {
    // Example usage:
    // java ProbabilityAggregator master.xlsx model1.xlsx model2.xlsx model3.xlsx
    if (args.length < 2) {
      System.out.println("Usage: java ProbabilityAggregator   ...");
      return;
    }

    String masterPath = args[0];

    // Load all workbooks
    Map sheets = new HashMap<>();
    Map books = new HashMap<>();

    for (int i = 0; i < args.length; i++) {
      String path = args[i];
      try (FileInputStream in = new FileInputStream(path)) {
        XSSFWorkbook wb = new XSSFWorkbook(in);
        XSSFSheet sh = wb.getSheetAt(0);
        String key = (i == 0) ? "MASTER" : ("MODEL" + i);
        sheets.put(key, sh);
        books.put(key, wb);
      }
    }

    XSSFSheet masterSheet = sheets.get("MASTER");

    // For each row, count models where Interest=="Yes" and write to column J (index 9)
    for (int rowNum = 1; rowNum <= masterSheet.getLastRowNum(); rowNum++) {
      int yesCount = 0;

      for (String name : sheets.keySet()) {
        if (name.equals("MASTER")) continue;
        XSSFRow row = sheets.get(name).getRow(rowNum);
        if (row == null) continue;

        XSSFCell cell = row.getCell(12); // Interest column
        if (cell != null && "Yes".equalsIgnoreCase(cell.getStringCellValue())) {
          yesCount++;
        }
      }

      XSSFRow outRow = masterSheet.getRow(rowNum);
      if (outRow == null) outRow = masterSheet.createRow(rowNum);
      XSSFCell outCell = outRow.createCell(9, CellType.STRING);
      outCell.setCellValue("Yes in " + yesCount + " models");
    }

    // Write master workbook back
    try (FileOutputStream out = new FileOutputStream(masterPath)) {
      books.get("MASTER").write(out);
    }

    // Close remaining workbooks
    for (Map.Entry e : books.entrySet()) {
      if (!e.getKey().equals("MASTER")) e.getValue().close();
    }
    books.get("MASTER").close();

    System.out.println("Aggregation complete.");
  }
}

This script consolidates results from multiple model runs. For each row, it counts how many models flagged it as “Interest = Yes”, then writes a compact summary back into a master file. The point is to quickly surface the most consistent contacts across runs without hand-checking each spreadsheet.

TB Drug-Synergy Screening Pipeline (INDIGO-style)

Automated modeling runs and ranked shortlists to prioritize combinations for follow-up testing.

MATLAB ETL batch correction (ComBat) ranking
  • Ran high-throughput prediction jobs screening 100,000+ drug combinations with repeatable logging
  • Generated analysis-ready Excel exports and ranked candidate lists for lab review
  • Computed correlation (Pearson/Spearman) and p-values to compare predictions across conditions
MATLAB pipeline overview (expand)
1) Append a new condition → impute → ComBat

Ingest a new transcriptome/fitness file, align by gene ID, fill missing values per-column (median fallback), and run ComBat to remove batch effects. Save a single corrected matrix for all downstream predictions.

2) Build phenotype features

Convert expression into a phenotype representation (z-thresholded) so the model can compare conditions consistently.

3) Train → predict → export ranked combinations

Train on curated interaction data, then run batched predictions for 2-way / 3-way / 4-way combinations. Export standardized tables for sorting and triage.

4) Correlation + p-values (reporting)

Compute Pearson/Spearman correlations between prediction vectors and report p-values to quantify agreement. Rank primarily by Spearman for robustness.
